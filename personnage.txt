package Position;

import java.util.ArrayList;
import java.util.List;
import java.util.Random;

public class personnage extends Position {
	Random random = new Random();

	public personnage (int[] coord) {
		super(coord);
	}

	public void deplacement(String cmd) {
			if(cmd.equals("z")) {
				ArrayList<Integer> Acoord = new ArrayList<Integer>();
				Acoord.add(coord[0]);
				Acoord.add(coord[1]+1);
				if (mouvPossible(plateau, Acoord) {
					coord[1]=coord[1]+1;
				}
			}
			else if(cmd.equals("q")) {
				ArrayList<Integer> Acoord = new ArrayList<Integer>();
				Acoord.add(coord[0]-1);
				Acoord.add(coord[1]);
				if (mouvPossible(plateau, Acoord) {
					coord[0]=coord[0]-1;
				}
			}
			else if(cmd.equals("s")) {
				ArrayList<Integer> Acoord = new ArrayList<Integer>();
				Acoord.add(coord[0]);
				Acoord.add(coord[1]-1);
				if (mouvPossible(plateau, Acoord) {
					coord[1]=coord[1]-1;
				}
			}
			else if(cmd.equals("d")) {
				ArrayList<Integer> Acoord = new ArrayList<Integer>();
				Acoord.add(coord[0]+1);
				Acoord.add(coord[1]);
				if (mouvPossible(plateau, Acoord) {
					coord[0]=coord[0]+1;
				}
			}
	}

public boolean mouvPossible(ArrayList<List<Integer>> plateau, ArrayList<Integer> coord) {
		for (List<Integer> element : plateau) {
			if (element.equals(coord)) {
				return true;
			}
		}
		return false;	
	}
}
