package Position;

import java.util.Random;

public class Monstre extends personnage {
	
	public Monstre (int[] coord) {
		super(coord);
	}
	
	Random rand = new Random();
	
	public void deplacement_monstre() {
		int r = rand.nextInt(5);
		String cmd = "";
		if (r == 0) {
			// Le monstre veut aller en haut
			cmd = "z";
			deplacement(cmd);
		}
		else if (r == 1) {
			// Le monstre veut aller à gauche
			cmd = "q";
			deplacement(cmd);
		}
		else if (r == 2) {
			// Le monstre veut aller en bas
			cmd = "s";
			deplacement(cmd);
		}
		else if (r == 3) {
			// Le monstre veut aller à droite
			cmd = "d";
			deplacement(cmd);
		}
		else {
			// Le monstre ne bouge pas
		}
	}
}
